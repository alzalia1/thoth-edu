/eval/get_eval

< retour
Obtiens les informations relatives à une eval pour élève

Status

✅ Terminé

Requête

                    
{
    "id" : "id de l'accès"
}

Réponse

                        
{
    "name" : "nom de l'accès",
    "eval" : {
        "name" : "nom de l'évaluation",
        "questions" : [ 
            {
                "id" : "id de la question",
                "type" : "traduction",
                "instruction" : "consigne",
                "points" : 0, // points de la question
            },
            ... 
            ],  
        },
    "time" : {
        "start" : "timestamp de début d'accès",
        "end" : "timestamp de fin d'accès"
        }
}
                        
                    
                        
{
    "name" : "nom de l'accès",
    "eval" : {
        "name" : "nom de l'évaluation",
        "questions" : [ 
            {
                "id" : "id de la question",
                "type" : "conjugaison",
                "instruction" : "consigne",
                "points" : 0, // points de la question
                "answer" : {
                    "tenses" : ["", ... ],
                    "pronouns": ["", ... ]
                }
            },
            ... 
            ],  
        },
    "time" : {
        "start" : "timestamp de début d'accès",
        "end" : "timestamp de fin d'accès"
        }
}
