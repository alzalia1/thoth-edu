/dashboard/eval/get

< retour
Renvoie une liste d'accès et d'autres informations relatives à une eval

Status

✅ Terminé

Requête

                    
{
    "id" : "id de l'eval"
}

Réponse

                    
{
    "name" : "nom de l'eval",
    "mark" : 0, // moyenne de l'évaluation
    "nb_ans" : 0, // nombre de réponses totales
    "access" : [ 
        { 
            "name" : "nom de l'accès", 
            "id" : "id de l'accès",
            "status" : 0, // 0 = Prévu ; 1 = En cours ; 2 = Terminé
            "mark" : 0, // moyenne de l'accès 
        },
        ... 
    ] 
}

Exemple

                            
{
    "id" : "31r2vm"
}
→
                            
{
    "name" : "Vocab - Septembre",
    "mark" : 12.6,
    "nb_ans" : 62,
    "access" : [ 
        { 
            "name" : "TG2", 
            "id" : "hl1kfbn4r",
            "status" : 2,
            "mark" : 12.4,
        },
        { 
            "name" : "TG3", 
            "id" : "w02fuh9d",
            "status" : 1,
            "mark" : 12.8,
        } 
    ] 
}
